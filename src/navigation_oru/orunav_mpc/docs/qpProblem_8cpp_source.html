279  unsigned int ind;

281  {

283  {
284  Sx0.multiply(ind, parameters.state_gains_mode2);
285  }
286  else
287  {
288  Sx0.multiply(ind, parameters.state_gains_mode1);
289  }
290  }
291 
292

294  fill_n(p.raw, SW_CONTROL_CONCAT_LEN, 0.0);
295 
296  // form transpose(Sx0)*T
297  // loop over columns (handle Nu=2 columns at each loop)
298  for (unsigned int i = 0; i < SW_PREVIEW_WINDOW_LEN; ++i)
299  {
300  ind = i*SW_CONTROL_VAR_N*SW_STATE_CONCAT_LEN;
301  for (unsigned int j=i*SW_STATE_VAR_N; j<SW_STATE_CONCAT_LEN; ++j)
302  {
303  p.v(i) += T[ind+j] * Sx0.raw[j];
304  }
305 
306  ind += SW_STATE_CONCAT_LEN;
307  for (unsigned int j=i*SW_STATE_VAR_N; j<SW_STATE_CONCAT_LEN; ++j)

323 void qpProblem::formHessian(
324  const swParameters &parameters,

328  int ind_xu1 = 0, ind_xu2 = 0;
329 
330  // loop over columns (handle Nu=2 columns at each loop)
331  State sqrt_state_gains;
332  for (unsigned int i = 0, k = 0; i < SW_PREVIEW_WINDOW_LEN; ++i)
333  {
334  ind_u1 = k*SW_STATE_CONCAT_LEN;
335  ++k;
336  ind_u2 = k*SW_STATE_CONCAT_LEN;
337  ++k;
338  // loop over rows (handle Nx=4 rows at each loop)
339  for (unsigned int j = i; j < SW_PREVIEW_WINDOW_LEN; ++j)
340  {
341  if (preview_win.mode[j + 1] == SW_STEP_MODE_2)
342  {
343  sqrt_state_gains = parameters.sqrt_state_gains_mode2;
344  }
345  else
346  {
347  sqrt_state_gains = parameters.sqrt_state_gains_mode1;
348  }
349 
350  ind_xu1 = SW_STATE_VAR_N*j + ind_u1;
351  ind_xu2 = SW_STATE_VAR_N*j + ind_u2;
352 
353  T[ind_xu1+0] *= sqrt_state_gains.x();
354  T[ind_xu1+1] *= sqrt_state_gains.y();
355  T[ind_xu1+2] *= sqrt_state_gains.theta();
356  T[ind_xu1+3] *= sqrt_state_gains.phi();
357 
358  T[ind_xu2+0] *= sqrt_state_gains.x();
359  T[ind_xu2+1] *= sqrt_state_gains.y();
360  T[ind_xu2+2] *= sqrt_state_gains.theta();
361  T[ind_xu2+3] *= sqrt_state_gains.phi();
362  }
363  }

367  Map< Matrix<double, SW_CONTROL_CONCAT_LEN, SW_CONTROL_CONCAT_LEN> > P_map(P);
368  P_map = T_map.transpose()*T_map;
369 
370  Control control_gains;

372  {
373  if (preview_win.mode[i/SW_CONTROL_VAR_N + 1] == SW_STEP_MODE_2)

376  }

380  }
381

384  P[i*SW_CONTROL_CONCAT_LEN + i] += control_gains.w();

386  }
