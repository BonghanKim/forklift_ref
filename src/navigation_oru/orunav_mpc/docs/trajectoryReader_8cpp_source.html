182 void trajectoryReader::processTrajectoryChunk_(
183  const orunav_msgs::ControllerTrajectoryChunk& msg)
184 {
185  // Correctness
186  if (msg.robot_id != thdata->parameters.robot_id)
187  {
188  return;
189  }
190 
191  if (msg.steps.size() == 0)
192  {
193  ROS_WARN("Trajectory message parser: Empty trajectory chunk.");
194  return;
195  }
196 
197 
198  try
199  {
200  SW_LOG_TRAJECTORY("ID = " << msg.traj_id << " | SEQ = " << msg.sequence_num);
201 
202 
203  // Initialize constraints
204  Constraints constraints(thdata->parameters);
205  initStepConstraints_(msg, constraints);
206 
207  trajectoryChunk traj_chunk(constraints);
208  traj_chunk.sequence_num = msg.sequence_num;
209  traj_chunk.id = msg.traj_id;
210  traj_chunk.steps.clear();
211 
212 
213  // Add steps
214  trajectoryStep step;
215  for (unsigned int i = 0; i < msg.steps.size(); ++i)
216  {
217  if (msg.steps[i].mode == msg.steps[i].MODE_1)
218  {
219  step.mode = SW_STEP_MODE_1;
220  }
221  else if (msg.steps[i].mode == msg.steps[i].MODE_2)
222  {
223  step.mode = SW_STEP_MODE_2;
224  }
225  else
226  {
227  ROS_WARN("Trajectory message parser: Unknown step mode, default is used.");
228  step.mode = SW_STEP_MODE_DEFAULT;
229  }
230 
231  step.state.set(
232  msg.steps[i].state.position_x,
233  msg.steps[i].state.position_y,
234  msg.steps[i].state.orientation_angle,
235  msg.steps[i].state.steering_angle);
236 
237  step.control.set(
238  msg.steps[i].velocities.tangential,
239  msg.steps[i].velocities.steering);
240 
241  traj_chunk.steps.push_back(step);
242  }
243 
244  // Finalize if necessary
245  if (msg.final == true)
246  {
247  traj_chunk.finalize();
248  }
249 
250  // Add
251  thdata->traj_pool.addTrajectoryChunk(traj_chunk);
252  }
253  catch (const exception &e)
254  {
255  SW_LOG_TRAJECTORY(e.what());
256  ROS_WARN("%s", e.what());
257  }
258 }

357 void trajectoryReader::actOnCommand(const orunav_msgs::ControllerCommandConstPtr& cmd)
358 {
359  if (cmd->robot_id != thdata->parameters.robot_id)
360  {
361  return;
362  }
363 
364  try
365  {
366  SW_LOG_TRAJECTORY("Command: " << cmd->command);
367 
368  if (cmd->command == cmd->COMMAND_BRAKE)
369  {
370  thdata->status.set(SW_PROCESS_STATUS_FAIL);
371  ROS_INFO("Command = brake: Stopping.");
372  }
373  else if (cmd->command == cmd->COMMAND_ACTIVATE)
374  {
375  thdata->traj_pool.setActive(cmd->traj_id);
376  ROS_INFO("Command = activate: Trajectory [%d] is activated.", cmd->traj_id);
377  }
378  else if (cmd->command == cmd->COMMAND_RECOVER)
379  {
380  thdata->status.recover();
381  ROS_INFO("Command = recover: Executed.");
382  }
383  else if (cmd->command == cmd->COMMAND_STARTTIME)
384  {
385  if (thdata->traj_pool.isBlocked() == false)
386  {
387  ROS_WARN("Command = starttime: Ignored, since the trajectory pool is already unblocked.");
388  }
389  else
390  {
391  start_time = cmd->start_time;
392  double timeDiff = (ros::Time::now() - start_time).toNSec() * 1e-6;
393  ROS_INFO("[orunav_mpc] Elapsed time since command: %3.3f",timeDiff);
394  if (start_time < ros::Time::now() - ros::Duration(SW_START_TIME_ALLOWED_ERROR_SEC))
395  {
396  start_time = ros::Time(0);
397  thdata->status.set(SW_PROCESS_STATUS_FAIL);
398  ROS_WARN("Command = starttime: The specified time is in the past.");
399  }
400  else
401  {
402  ROS_INFO("Command = starttime: Time is set, waiting.");
403  }
404  }
405  }
406  else
407  {
408  ROS_WARN("Command = ?: An unsupported command is received.");
409  }
410  }
411  catch (const exception &e)
412  {
413  SW_LOG_TRAJECTORY(e.what());
414  ROS_WARN("%s", e.what());
415  }
416 }
