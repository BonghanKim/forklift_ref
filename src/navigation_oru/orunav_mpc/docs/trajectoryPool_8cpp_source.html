218 
219  for(;;)

222  while (local_index.step >= traj[local_index.chunk].steps.size())

234

370 void trajectoryPool::addTrajectoryChunk(const trajectoryChunk & traj_chunk)
371 {
372  swMutexScopedLock traj_lock(mutex);
373 
374  map<trajectoryID, deque<trajectoryChunk> >::iterator it = traj_pool.find(traj_chunk.id);
375 
376  if (it == traj_pool.end())
377  {
378  if (traj_chunk.sequence_num == 0)
379  {
380  // add trajectory
381  traj_pool[traj_chunk.id].push_back(traj_chunk);
382  }
383  else
384  {
385  SW_THROW_MSG("Attempt to append an unknown trajectory.");
386  }
387  }
388  else
389  {
390  deque<trajectoryChunk> &traj = it->second;
391 
392  if ((traj.front().sequence_num <= traj_chunk.sequence_num) &&
393  (traj.back().sequence_num >= traj_chunk.sequence_num))
394  {
395  // replace
396  unsigned int chunk_index = traj_chunk.sequence_num - traj.front().sequence_num;
397  traj[chunk_index] = traj_chunk;
398 
399  if (traj_chunk.finalized) {
400  // clear the remaining part of the traj_chunks
401  traj.erase(traj.begin()+chunk_index+1, traj.end());
402  }
403  }
404  else if (traj_chunk.sequence_num == traj.back().sequence_num + 1)
405  {
406  if (traj.back().finalized)
407  {
408  SW_THROW_MSG("Attempt to append a finalized trajectory.");
409  }
410  else
411  {
412  // append
413  traj.push_back(traj_chunk);
414  }
415  }
416  else
417  {
418  // Wrong sequence number
419  SW_THROW_MSG("Wrong sequence number of a trajectory chunk.");
420  }
421  }
422 }
